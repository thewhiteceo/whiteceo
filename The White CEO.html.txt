<script>
let currentMode = 'Prompt';

function setMode(mode, e){
  currentMode = mode;
  document.querySelectorAll('.btn').forEach(b => b.classList.remove('active'));
  e.target.classList.add('active');
  document.getElementById('output').innerHTML = '';
}

function generate(){
  let input = document.getElementById('input').value;
  let output = document.getElementById('output');
  
  if(currentMode === 'Prompt'){
    output.innerHTML = `PROMPT MODE:\n\nYou asked: "${input}"\n\nAI Response: I will help you build this!`;
  }
  if(currentMode === 'Code'){
    output.innerHTML = `CODE MODE:\n\nHere is code for: "${input}"\n\n<html>\n<head><title>Login</title></head>\n<body>\n<form>\nEmail: <input type="email"><br>\nPassword: <input type="password"><br>\n<button>Login</button>\n</form>\n</body>\n</html>`;
  }
  if(currentMode === 'Preview'){
    output.innerHTML = `PREVIEW MODE:\n\n${input}\n\n[This will show live preview of your code]`;
  }
  if(currentMode === 'Image'){
    output.innerHTML = `IMAGE MODE:\n\nGenerating image for: "${input}"\n\n[Image AI coming soon 👑]`;
  }
  if(currentMode === 'Assist'){
    output.innerHTML = `ASSIST MODE:\n\nLet me help you with: "${input}"\n\nTips:\n1. Break it into small parts\n2. Use AI to debug\n3. Test step by step`;
  }
  if(currentMode === 'Hacker'){
    output.innerHTML = `HACKER MODE:\n\nAnalyzing: "${input}"\n\nSecurity tips:\n- Use HTTPS\n- Validate inputs\n- Hash passwords\n[Ethical use only] 👑`;
  }
}
</script>
